// Firestore Security Rules
// 去 Firebase Console → Firestore Database → 規則貼上以下內容

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // 評分系統 - 任何人可讀，登入用戶可寫
    match /ratings/{ratingId} {
      allow read: if true;
      allow create: if request.auth != null && 
        request.auth.uid == request.resource.data.userId;
      allow update, delete: if request.auth != null && 
        request.auth.uid == resource.data.userId;
    }
    
    // 用戶歌單 - 公開可讀，只有創建者可修改
    match /userPlaylists/{playlistId} {
      allow read: if true;
      allow create: if request.auth != null && 
        request.auth.uid == request.resource.data.userId;
      allow update, delete: if request.auth != null && 
        request.auth.uid == resource.data.userId;
    }
    
    // 喜愛歌曲 - 只有本人可讀寫
    match /userLikedSongs/{docId} {
      allow read: if request.auth != null && 
        request.auth.uid == resource.data.userId;
      allow create: if request.auth != null && 
        request.auth.uid == request.resource.data.userId;
      allow delete: if request.auth != null && 
        request.auth.uid == resource.data.userId;
    }
    
    // 歌曲資料 - 允許更新評分欄位，所有登入用戶可寫（Admin 權限在前端控制）
    match /tabs/{tabId} {
      allow read: if true;
      allow create, update, delete: if request.auth != null;
    }
    
    // Artists collection - 所有登入用戶可寫（Admin 權限在前端控制）
    match /artists/{artistId} {
      allow read: if true;
      allow write: if request.auth != null;
    }
    
    // Users collection - 用戶可寫自己的資料，Admin 可寫所有（前端控制）
    match /users/{userId} {
      allow read: if request.auth != null;
      allow write: if request.auth != null;
    }
    
    // Comments collection
    match /comments/{commentId} {
      allow read: if true;
      allow create: if request.auth != null;
      allow delete: if request.auth != null && 
        request.auth.uid == resource.data.userId;
    }
    
    // TabRequests collection
    match /tabRequests/{requestId} {
      allow read: if true;
      allow create: if request.auth != null;
      allow update: if request.auth != null;
    }
    
    // System Playlists collection - 所有登入用戶可寫（Admin 權限在前端控制）
    match /playlists/{playlistId} {
      allow read: if true;
      allow create, update, delete: if request.auth != null;
    }
    
    // Settings collection - 所有登入用戶可寫（Admin 權限在前端控制）
    match /settings/{settingId} {
      allow read: if true;
      allow write: if request.auth != null;
    }
    
    // Page Views collection
    match /pageViews/{viewId} {
      allow create: if true;
      allow read: if true;
      allow update, delete: if false;
    }
    
    // Daily Stats collection - 所有登入用戶可讀寫（Admin 權限在前端控制）
    match /dailyStats/{dateId} {
      allow read, write: if request.auth != null;
    }
    
    // Recent Views collection
    match /recentViews/{docId} {
      allow read: if request.auth != null && 
        request.auth.uid == resource.data.userId;
      allow create: if request.auth != null;
      allow update, delete: if request.auth != null && 
        request.auth.uid == resource.data.userId;
    }
    
    // Nav Icons collection
    match /navIcons/{docId} {
      allow read: if true;
      allow write: if request.auth != null;
    }
  }
}
