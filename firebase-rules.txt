// Firestore Security Rules
// 去 Firebase Console → Firestore Database → 規則貼上以下內容

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Tabs collection
    match /tabs/{tabId} {
      // 所有人可以讀取
      allow read: if true;
      
      // 登入用戶可以建立
      allow create: if request.auth != null;
      
      // 更新和刪除權限：
      // - 新譜（有 createdBy）：只有上傳者或 Admin 可以編輯/刪除
      // - 舊譜（沒有 createdBy）：只有 Admin 可以編輯/刪除
      allow update, delete: if request.auth != null 
        && (
          // Admin 可以編輯/刪除所有譜
          request.auth.token.email == 'kermit.tam@gmail.com'
          || 
          // 新譜：上傳者可以編輯/刪除
          (resource.data.keys().hasAny(['createdBy']) && resource.data.createdBy == request.auth.uid)
        );
    }
    
    // Artists collection
    match /artists/{artistId} {
      // 所有人可以讀取
      allow read: if true;
      
      // 登入用戶可以寫入
      allow write: if request.auth != null;
    }
    
    // Users collection
    match /users/{userId} {
      // 所有人可以讀取
      allow read: if true;
      
      // 只有自己可以寫入自己嘅資料
      allow write: if request.auth != null 
        && request.auth.uid == userId;
    }
    
    // Comments collection（樂譜留言）
    match /comments/{commentId} {
      // 所有人可以讀取
      allow read: if true;
      
      // 登入用戶可以建立
      allow create: if request.auth != null;
      
      // 只有留言者自己可以刪除
      allow delete: if request.auth != null 
        && request.auth.uid == resource.data.userId;
    }
    
    // TabRequests collection（求譜）
    match /tabRequests/{requestId} {
      // 所有人可以讀取
      allow read: if true;
      
      // 登入用戶可以建立
      allow create: if request.auth != null;
      
      // 只有管理員或建立者可以更新狀態
      allow update: if request.auth != null;
    }
    
    // Playlists collection（歌單）
    match /playlists/{playlistId} {
      // 所有人可以讀取
      allow read: if true;
      
      // 登入用戶可以建立和更新（管理員功能）
      allow create, update: if request.auth != null;
      
      // 只有管理員可以刪除
      allow delete: if request.auth != null;
    }
    
    // Settings collection（網站設定，包括 Logo）
    match /settings/{settingId} {
      // 所有人可以讀取
      allow read: if true;
      
      // 登入用戶可以更新（管理員功能）
      allow write: if request.auth != null;
    }
    
    // PageViews collection（頁面瀏覽統計）
    match /pageViews/{viewId} {
      // 允許任何人寫入（用於追蹤頁面瀏覽）
      allow create: if true;
      
      // 只有管理員可以讀取統計數據
      allow read: if request.auth != null 
        && request.auth.token.email == 'kermit.tam@gmail.com';
      
      // 不允許更新或刪除
      allow update, delete: if false;
    }
  }
}
